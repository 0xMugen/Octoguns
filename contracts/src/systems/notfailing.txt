
fn shoot(world: IWorldDispatcher, ref positions: Array<CharacterPosition>, ref bullets: Array<Bullet>, ref shot: Shot, step: u16) {
    for character in positions {
        let bullet = BulletTrait::new(
            world.uuid(), 
            Vec2 {x: character.coords.x, y: character.coords.y}, 
            shot.angle, 
            character.id,
            step
        );
        bullets.append(bullet);
        set!(world, (bullet));
    }
}

if sub_move_index == next_shot_t1.into() {
    //check for shots
    let mut shot = m1.shots.pop_front().unwrap();
    shoot(world, ref m1_positions, ref bullets, ref shot, step.try_into().unwrap());                    
}
